cmake_minimum_required(VERSION 3.10)
project(luxury_crm LANGUAGES CXX)

set(BINARY_NAME "luxury_crm")
set(APPLICATION_ID "com.luxury.crm")

cmake_policy(SET CMP0063 NEW)

set(CMAKE_INSTALL_RPATH "$ORIGIN/lib")

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

set(FLUTTER_MANAGED_DIR "${CMAKE_CURRENT_SOURCE_DIR}/flutter")

set(FLUTTER_ROOT "~/flutter")
set(FLUTTER_EPHEMERAL_DIR "${CMAKE_CURRENT_SOURCE_DIR}/ephemeral")

set(PROJECT_DIR "${CMAKE_CURRENT_SOURCE_DIR}")
set(FLUTTER_LIBRARY "${FLUTTER_EPHEMERAL_DIR}/libflutter_linux_gtk.so")
set(FLUTTER_ICU_DATA_FILE "${FLUTTER_EPHEMERAL_DIR}/icudtl.dat")

set(PLUGIN_BUNDLED_LIBRARIES)

include("${FLUTTER_MANAGED_DIR}/generated_plugins.cmake")

list(APPEND FLUTTER_LIBRARY_HEADERS
  "${FLUTTER_EPHEMERAL_DIR}/flutter_linux.h"
)

add_library(flutter INTERFACE)
target_include_directories(flutter INTERFACE
  "${FLUTTER_EPHEMERAL_DIR}"
)
target_link_libraries(flutter INTERFACE "${FLUTTER_LIBRARY}")

set(FLUTTER_BUNDLE_DIR "${PROJECT_BINARY_DIR}/bundle")
set(FLUTTER_LIBRARY_DIR "${FLUTTER_BUNDLE_DIR}/lib")

set(INSTALL_BUNDLE_DATA_DIR "${CMAKE_INSTALL_PREFIX}/data")
set(INSTALL_BUNDLE_LIB_DIR "${CMAKE_INSTALL_PREFIX}/lib")

install(TARGETS ${BINARY_NAME} RUNTIME DESTINATION "${CMAKE_INSTALL_PREFIX}"
  COMPONENT Runtime)

install(FILES "${FLUTTER_ICU_DATA_FILE}" DESTINATION "${INSTALL_BUNDLE_DATA_DIR}"
  COMPONENT Runtime)

install(FILES "${FLUTTER_LIBRARY}" DESTINATION "${INSTALL_BUNDLE_LIB_DIR}"
  COMPONENT Runtime)

if(PLUGIN_BUNDLED_LIBRARIES)
  install(FILES "${PLUGIN_BUNDLED_LIBRARIES}"
    DESTINATION "${INSTALL_BUNDLE_LIB_DIR}"
    COMPONENT Runtime)
endif()

set(FLUTTER_ASSETS_DIR "${FLUTTER_BUNDLE_DIR}/data/flutter_assets")
